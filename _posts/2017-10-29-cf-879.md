//codeforces 879a

// Round #443 Div.2

#include <cstdio>

#include <cmath>

#include <algorithm>

using namespace std;
int ans,b,flag,si,di;
int main(){
	int n=0;
	scanf("%d",&n);
	ans=0;
	for(int i=1;i<=n;i++){
		
		scanf("%d%d",&si,&di);
		/*if(si>ans)
			ans=si;
		printf("aans=%d\n",ans);*/
		flag=0;
		b=0;
		while(flag<1){
			if(si+di*b>ans){
				ans=si+di*b;
				flag++;
				//break;
			}
			else
				b++;
			//printf("ans=%d,b=%d\n",ans,b);
			
		}
	}
	printf("%d",ans);
}
